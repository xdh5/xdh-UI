<template>
  <button
    :class="['x-button', `icon-position-${position}`]"
    @click="$emit('click')"
    :disabled="disabled"
  >
    <x-icon
      class="x-icon"
      v-if="icon && !loading"
      :name="icon"
    ></x-icon>
    <x-icon
      v-if="loading"
      name="loading"
    ></x-icon>
    <slot></slot>
  </button>
</template>
<script>
import Icon from '../icon'
export default {
  name: 'x-button',
  components: {
    'x-icon': Icon
  },
  props: {
    icon: {},
    position: {
      type: String,
      default: 'left',
      validator: val => {
        return !(val !== 'left' && val !== 'right')
      }
    },
    loading: {
      type: Boolean,
      default: false
    },
    disabled: {
      type: Boolean,
      default: false
    }
  }
}
</script>
