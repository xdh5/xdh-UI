<template>
  <div class="x-nav-item" :class="{'active':active === name}" @click="onClick"
    :data-name="name">
    <slot></slot>
  </div>
</template>
 
<script>
export default {
  name: 'x-nav-item',
  inject: ['eventBus'],
  props: {
    name: {
      type: String,
      required: true
    }
  },
  data() {
    return {
      active: undefined
    }
  },
  methods: {
    onClick() {
      this.eventBus.$emit('update', this.name)
    }
  },
  mounted() {
    this.eventBus.$on('selected', active => {
      this.active = active
    })
  }
}
</script>
